async function sendTelegram(isTest=false){

  const token=els.botToken.value.trim(),chatId=els.chatId.value.trim();

  if(!token||!chatId){if(isTest)throw new Error('Token oder Chat-ID fehlt');els.telegramStatus.textContent='Telegram nicht konfiguriert.';return;}

  const coolMs=Number(els.telegramCooldown.value||10)*60000;

  const now=Date.now();

  if(!isTest&&now-state.lastTelegramAt<coolMs)return;

  const text=`${els.telegramMessage.value.trim()||'🐱 Katze am Trinken erkannt.'}\n${new Date().toLocaleString('de-DE')}`;

  if(els.sendPhoto.checked){

    const blob=await snapshotBlob();

    const form=new FormData();form.append('chat_id',chatId);form.append('caption',text);form.append('photo',blob,'katze.jpg');

    const res=await fetch(`https://api.telegram.org/bot${token}/sendPhoto`,{method:'POST',body:form});

    const data=await res.json();if(!data.ok)throw new Error(data.description||'sendPhoto fehlgeschlagen');

  }else{

    const res=await fetch(`https://api.telegram.org/bot${token}/sendMessage`,{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({chat_id:chatId,text})});

    const data=await res.json();if(!data.ok)throw new Error(data.description||'sendMessage fehlgeschlagen');

  }

  state.lastTelegramAt=now;

  els.telegramStatus.innerHTML='<span class="ok">✓ Telegram gesendet.</span>';

  saveLocal();toast(isTest?'Telegram-Test erfolgreich.':'Telegram-Alarm gesendet.');

}

function scheduleStop(){if(state.recordStopTimer)clearTimeout(state.recordStopTimer);state.recordStopTimer=setTimeout(()=>stopRecording('Nachlauf'),Number(els.postRoll.value||8)*1000);}

function stopRecording(reason=''){if(!state.isRecording||!state.recorder)return;try{state.recorder.stop();}catch(e){}state.isRecording=false;updateBadges();if(reason)toast('Clip beendet: '+reason);}

async function startRecording(motion){

  if(!state.stream)return;

  const now=Date.now(),coolMs=Number(els.cooldown.value||12)*1000;

  if(state.isRecording){scheduleStop();return;}

  if(now-state.lastTriggerAt<coolMs)return;

  state.lastTriggerAt=now;state.recorderChunks=[];

  const mt=mimeType();

  state.recorder=new MediaRecorder(state.stream,mt?{mimeType:mt}:undefined);

  state.isRecording=true;updateBadges();

  state.recorder.ondataavailable=ev=>{if(ev.data&&ev.data.size>0)state.recorderChunks.push(ev.data);};

  state.recorder.onstop=()=>{const blob=new Blob(state.recorderChunks,{type:mt||'video/webm'});if(blob.size)addClip(blob,motion);state.recorder=null;state.recorderChunks=[];};

  state.recorder.start(1000);scheduleStop();

  if(state.clipTimer)clearTimeout(state.clipTimer);

  state.clipTimer=setTimeout(()=>stopRecording('Max. Länge'),Number(els.clipLimit.value||25)*1000);

  try{await sendTelegram(false);}catch(e){els.telegramStatus.textContent='Telegram-Fehler: '+e.message;}

  toast('Bewegung erkannt – Aufnahme gestartet.');

}

function detectMotion(){

  if(!state.monitoring||!state.stream||!els.video.videoWidth)return;

  const vw=els.video.videoWidth,vh=els.video.videoHeight;

  const sW=320,sH=Math.round(vh*(sW/vw));

  els.work.width=sW;els.work.height=sH;

  wctx.drawImage(els.video,0,0,sW,sH);

  const cur=wctx.getImageData(0,0,sW,sH);

  let level=0;

  const r=roi(),x0=Math.max(0,Math.floor(r.x*sW)),y0=Math.max(0,Math.floor(r.y*sH)),x1=Math.min(sW,Math.floor((r.x+r.w)*sW)),y1=Math.min(sH,Math.floor((r.y+r.h)*sH));

  const thresh=Number(els.sensitivity.value||18)*3;

  if(state.lastFrameData){

    const a=cur.data,b=state.lastFrameData.data;

    for(let y=y0;y<y1;y+=2)for(let x=x0;x<x1;x+=2){const i=(y*sW+x)*4;if(Math.abs(a[i]-b[i])+Math.abs(a[i+1]-b[i+1])+Math.abs(a[i+2]-b[i+2])>thresh)level++;}

  }

  state.lastFrameData=cur;

  const scaled=level*4;const hit=scaled>=Number(els.motionPixels.value||1400);

  els.motionLevel.textContent=String(scaled);drawOverlay(scaled,hit);

  if(hit){els.lastMotion.textContent=fmtDate(Date.now());startRecording(scaled);}

  state.motionLoop=requestAnimationFrame(detectMotion);

}
